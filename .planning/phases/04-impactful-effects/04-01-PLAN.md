The final method body should be:

```javascript
generateParticleTextures() {
  const g = this.make.graphics({ add: false });
  g.fillStyle(0xffffff, 1);
  g.fillCircle(2, 2, 2);
  g.generateTexture('particle', 4, 4);
  g.clear();

  g.fillCircle(4, 4, 4);
  g.fillStyle(0xffffff, 1);
  g.fillCircle(4, 4, 2);
  g.generateTexture('particle-glow', 8, 8);
  g.destroy();
}
```
